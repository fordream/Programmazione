\documentclass[a4paper, 11pt]{article}
\usepackage{bm}
\usepackage{amsmath}
\usepackage{amssymb}
\newcommand{\disp}{\mbox{lolcosa}}
\newcommand{\dasp}{\text{lolcosa}}

\title{Prova di mbox e text}

\begin{document}
\maketitle

\begin{itemize}
	\item[normale mbox] $f(x) = \disp$
	\item[normale text] $f(x) = \dasp$
	\item[superscript mbox] $x^{\disp}$
	\item[superscript text] $x^{\dasp}$
	\item[italic mbox] {\it $f(x) = \disp$}
	\item[italic text] {\it $f(x) = \dasp$}
	\item[bold mbox] {\bm{$f(x) = \disp$}}
	\item[bold text] {\bm{$f(x) = \dasp$}}
\end{itemize}

\end{document}
